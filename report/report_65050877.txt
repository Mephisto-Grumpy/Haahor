SELECT D.DORM_ID, D.NAME, Z.ZONE_ID, Z.NAME, D.LATITUDE, D.LONGITUDE,
       D.PHONE_NUMBER, D.RATING_SCORE,
       F.FACILITY_NAME, F.ADD_ON,
       R.ROOM_TYPE, R.ROOM_PRICE, ZL.LANDMARK
FROM DORMITORY D
INNER JOIN FACILITY F ON D.DORM_ID = F.DORM_ID
INNER JOIN ROOM R ON D.DORM_ID = R.DORM_ID
INNER JOIN Zone Z ON D.ZONE_ID = Z.ZONE_ID
INNER JOIN Zone_Landmark ZL ON Z.ZONE_ID = ZL.ZONE_ID
INNER JOIN Review RV ON D.DORM_ID = RV.DORM_ID
INNER JOIN Rating RA ON RV.REVIEW_ID = RA.REVIEW_ID
WHERE R.ROOM_TYPE = 'ห้องแอร์'
  AND R.ROOM_PRICE <= 5500
  AND F.FACILITY_NAME = 'CCTV'
  AND ZL.LANDMARK = 'สถานบันเทิง'
GROUP BY D.DORM_ID, D.NAME, Z.ZONE_ID, Z.NAME, D.LATITUDE, D.LONGITUDE,
         D.PHONE_NUMBER, D.RATING_SCORE, F.FACILITY_NAME, F.ADD_ON,
         R.ROOM_TYPE, R.ROOM_PRICE, ZL.LANDMARK;
